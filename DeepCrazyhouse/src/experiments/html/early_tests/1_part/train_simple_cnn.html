def get_numpy_arrays(pgn_dataset):
    # Get the data
    x_train = list(pgn_dataset['x_train'])
    x_test = list(pgn_dataset['x_test'])
    y_train = list(pgn_dataset['y_train'])
    y_test = list(pgn_dataset['y_test'])
    
    x_train = np.stack(x_train, axis=0)
    x_test = np.stack(x_test, axis=0)
    y_train = np.stack(y_train, axis=0)
    y_test = np.stack(y_test, axis=0)
    
    return (x_train, y_train), (x_test, y_test)

def tiny_vgg(input_shape, out_number=1):

    model = Sequential()
    # Block 1
    model.add(Conv2D(32, kernel_size=(3, 3), activation='relu', input_shape=input_shape, padding='same'))
    model.add(Conv2D(32, (3, 3), activation='relu', padding='same'))

    # Block 2
    model.add(Conv2D(64, (3, 3), activation='relu', padding='same'))
    model.add(Conv2D(64, (3, 3), activation='relu', padding='same'))

    # Fully Connected
    model.add(Flatten())
    #model.add(Dropout(0.25))
    model.add(Dense(256, activation='relu'))
    #model.add(Dropout(0.5))
    #model.add(Dense(out_number, activation='tanh'))
    model.add(Dense(out_number, activation='softmax'))

    return model
