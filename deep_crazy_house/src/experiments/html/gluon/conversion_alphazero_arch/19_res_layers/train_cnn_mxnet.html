# https://github.com/keras-team/keras/blob/master/keras/metrics.py
def acc_sign(y_true, y_pred):
    if mode == 'mxnet':
        return (np.sign(y_pred).flatten() == y_true).sum() / len(y_true)
    elif mode == 'keras':
        return K.mean(K.equal(K.sign(y_pred), y_true), axis=-1)

def acc_round(y_true, y_pred):
    # each interval for class -1, 0, 1 must have a width of 2/3 to be equally distributed
    # therefore you must convert (1/3) to (1/2) and only then round to the closest integer
    # 1/3 * x = 1/2
    return K.mean(K.equal(K.round(y_pred * 1.5), y_true), axis=-1)

def evaluate_metrics(metrics, data_iterator, net, nb_batches=None):
    reset_metrics(metrics)    
    for i, (data, value_label, policy_label) in enumerate(data_iterator):
        data = data.as_in_context(ctx)
        value_label = value_label.as_in_context(ctx)
        policy_label = policy_label.as_in_context(ctx)
        
        [value_out, policy_out] = net(data)
            
        # update the metrics
        metrics['value_loss'].update(preds=value_out, labels=value_label)
        
        #print(policy_out.shape)
        #print(policy_label.shape)
        
        metrics['policy_loss'].update(preds=nd.SoftmaxActivation(policy_out), labels=policy_label)
        metrics['value_acc_sign'].update(preds=value_out, labels=value_label)
        metrics['policy_acc'].update(preds=nd.argmax(policy_out, axis=1), labels=policy_label)
        
        # stop after evaluating x batches (only recommeded to use this for the train set evaluation)
        if i is not None and i == nb_batches:
            break
            
    metric_values = {}
    metric_values['loss'] = 0.01 * metrics['value_loss'].get()[1] + 0.99 * metrics['policy_loss'].get()[1]
    
    for metric in metrics.values():
        metric_values[metric.get()[0]] = metric.get()[1]
    
    return metric_values

def reset_metrics(metrics):

def log_metrics(metric_values, global_step, prefix='train_'):
    for name in metric_values.keys():                  # show the metric stats
        print(' - %s%s: %.4f' % (prefix, name, metric_values[name]), end="")
        # add the metrics to the tensorboard event file
        sw.add_scalar(name, [prefix.replace('_', ''), metric_values[name]], global_step)

class PGNDataset(RecordFileDataset):

    def __init__(self, filename, input_shape, normalize):
        super(PGNDataset, self).__init__(filename)
        self.input_shape = input_shape
        self.input_shape_flatten = input_shape[0] * input_shape[1] * input_shape[2]        
        self.normalize = normalize
        
    def __getitem__(self, idx):
        item = self._record.read_idx(self._record.keys[idx])
        
        header, s = mx.recordio.unpack(item)    
        buf = zlib.decompress(s)
        
        data = np.frombuffer(buf, dtype=np.int16)
    
        x = data[:self.input_shape_flatten]
        x = x.reshape(self.input_shape)
        x = x.astype(np.float32)
            
        if self.normalize is True:
            normalize_input_planes(x)
        
        yv = header[1][0]
        yp = header[1][1]

        return x, yv, yp

t_s = time()

val_loss_best = None
epoch_best = None
val_p_acc_best = None

for e in range(epochs):
    batch_steps = 0
    t_e_s = time()
    
    for (data, value_label, policy_label) in tqdm_notebook(train_data):
        batch_steps += 1
        data = data.as_in_context(ctx)
        value_label = value_label.as_in_context(ctx)
        policy_label = policy_label.as_in_context(ctx)
        
        with autograd.record():
            
            [value_out, policy_out] = net(data)
            value_loss = l2_loss(value_out, value_label)
            policy_loss = softmax_cross_entropy(policy_out, policy_label)
            combined_loss = 0.01 * value_loss.sum() + 0.99 * policy_loss.sum()

            # update a dummy metric to see a proper progress bar
            metrics['value_loss'].update(preds=value_out, labels=value_label)
            
        combined_loss.backward()
        trainer.step(data.shape[0])
        
    ms_step = ((time() - t_e_s) / batch_steps) * 1000
        
    logging.info("Epoch %d/%d - %dms/step", e+1, epochs, ms_step)
    logging.info("-------------------------")
    
    # add the graph representation of the network to the tensorboard log file
    if e == 0:
        sw.add_graph(net)

    grads = []
    # logging the gradients of parameters for checking convergence
    for i, name in enumerate(param_names):
            if 'bn' not in name and 'batch' not in name:
                grads.append(params[name].grad())
                sw.add_histogram(tag=name, values=grads[-1], global_step=e, bins=20)
    
    train_metric_values = evaluate_metrics(metrics, train_data, net, nb_batches=25)
    log_metrics(train_metric_values, global_step=e+1, prefix='train_')
    
    val_metric_values = evaluate_metrics(metrics, val_data, net, nb_batches=None)
    log_metrics(val_metric_values, global_step=e+1, prefix='val_')
    
    # check if a new checkpoint shall be created
    if val_loss_best is None or val_metric_values['loss'] < val_loss_best:
        # update val_loss_best
        val_loss_best = val_metric_values['loss']
        val_p_acc_best = val_metric_values['policy_acc']
        epoch_best = e+1
        if export_model is True:
            prefix ="./weights/model-%.5f-%.3f-%04d.params" % (val_loss_best, val_p_acc_best, epochs)
            net.save_parameters(prefix)
        
    print(' - %.2fs'%(time() - t_e_s), end='')
    
print('Elapsed time for training(hh:mm:ss): ' + str(datetime.timedelta(seconds=round(time() - t_s))))

def predict_single(net, x):
    
    if mode == 'keras':
        pred = net.predict(np.expand_dims(x, axis=0))
    elif mode == 'mxnet':
        out = [None, None]
        pred = net(mx.nd.array(np.expand_dims(x, axis=0), ctx=ctx))
        pred[1] = pred[1].softmax()
        #pred = mod.forward(mx.nd.array(np.expand_dims(x, axis=0)), is_train=False)
        out[0] = pred[0].asnumpy()
        out[1] = pred[1].asnumpy()
    
    return out

def eval_pos(net, x_mate, yp_mate, verbose=False):
    
    board = get_board_of_current_mat(x_mate, normalized_input=normalize)
    if verbose is True:
        print("{0}'s turn".format(chess.COLOR_NAMES[board.turn]))
        print("black/white {0}".format(board.pockets))
    pred = predict_single(net, x_mate)

    true_mat = yp_mate.reshape(78,8,8)
    true_move = filter_moves(board, true_mat, is_white_to_move=board.turn, renormalize_confidences=False)[0][0]
    
    pred_mat = pred[1][0].reshape(78,8,8)
    opts = 5
    pred_moves = filter_moves(board, pred_mat, is_white_to_move=board.turn, renormalize_confidences=False)[0][:opts]

    legal_move_cnt = board.legal_moves.count()
    mate_move_cnt = str(board.legal_moves).count('#')
    
    is_mate_5_top = False
    
    for pred_move in pred_moves:
        board_5_top = deepcopy(board)
        board_5_top.push(pred_move)
        if board_5_top.is_checkmate() is True:
            is_mate_5_top = True
            break
    
    board.push(pred_moves[0])
    
    is_checkmate = False
    if board.is_checkmate() is True:
        is_checkmate = True
        
    filtered_pred = sorted(pred[1][0], reverse=True)
    
    if verbose is True:
        plt.barh(range(opts)[::-1], filtered_pred[:opts])
        ax = plt.gca()
        ax.set_yticks(range(opts)[::-1])
        ax.set_yticklabels(pred_moves)
        plt.title('True Move:' + str(true_move) +
                 '\nEval:' + str(pred[0][0]))
        plt.show()
    
    return pred, pred_moves, true_move, board, is_checkmate, is_mate_5_top, legal_move_cnt, mate_move_cnt
