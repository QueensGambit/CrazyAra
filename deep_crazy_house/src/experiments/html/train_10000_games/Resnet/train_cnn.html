def acc_sign(y_true, y_pred):
    return K.mean(K.equal(K.sign(y_pred), y_true), axis=-1)

def acc_round(y_true, y_pred):
    # each interval for class -1, 0, 1 must have a width of 2/3 to be equally distributed
    # therefore you must convert (1/3) to (1/2) and only then round to the closest integer
    # 1/3 * x = 1/2
    return K.mean(K.equal(K.round(y_pred * 1.5), y_true), axis=-1)

def eval_pos(x_mate, yp_mate):
    
    board = get_mat_of_current_board(x_mate)
    print("{0}'s turn".format(chess.COLOR_NAMES[board.turn]))
    print("black/white {0}".format(board.pockets))
    pred = model.predict(np.expand_dims(x_mate, axis=0))

    true_mat = yp_mate.reshape(78,8,8)
    true_move = filter_moves(board, true_mat, is_white_to_move=board.turn)[0][0]
    
    pred_mat = pred[1][0].reshape(78,8,8)
    opts = 5
    pred_moves = filter_moves(board, pred_mat, is_white_to_move=board.turn)[0][:opts]

    # somehow we must mirror the move for white ?!
    if board.turn == chess.WHITE:
        true_move = mirror_move(true_move)
        for i, pred_move in enumerate(pred_moves):
            pred_moves[i] = mirror_move(pred_move)
        
    filtered_pred = sorted(pred[1][0], reverse=True)
    
    plt.barh(range(opts)[::-1], filtered_pred[:opts])
    ax = plt.gca()
    ax.set_yticks(range(opts)[::-1])
    ax.set_yticklabels(pred_moves)
    plt.title('True Move:' + str(true_move) +
             '\nEval:' + str(pred[0][0]))
    plt.show()
    
    return pred, pred_moves, true_move, board
