def get_numpy_arrays(pgn_dataset):
    # Get the data
    x_train = list(pgn_dataset['x_train'])
    x_test = list(pgn_dataset['x_test'])
    yv_train = list(pgn_dataset['yv_train'])
    yv_test = list(pgn_dataset['yv_test'])
    yp_train = list(pgn_dataset['yp_train'])
    yp_test = list(pgn_dataset['yp_test'])
    
    x_train = np.stack(x_train, axis=0)
    x_test = np.stack(x_test, axis=0)
    yv_train = np.stack(yv_train, axis=0)
    yv_test = np.stack(yv_test, axis=0)
    yp_train = np.stack(yp_train, axis=0)
    yp_test = np.stack(yp_test, axis=0)    
    
    return (x_train, yv_train, yp_train), (x_test, yv_test, yp_test)

# https://machinelearningmastery.com/custom-metrics-deep-learning-keras-python/
# https://github.com/keras-team/keras/blob/master/keras/metrics.py
def acc_sign(y_true, y_pred):
    return K.mean(K.equal(K.sign(y_pred), y_true), axis=-1)

def acc_round(y_true, y_pred):
    # each interval for class -1, 0, 1 must have a width of 2/3 to be equally distributed
    # therefore you must convert (1/3) to (1/2) and only then round to the closest integer
    # 1/3 * x = 1/2
    return K.mean(K.equal(K.round(y_pred * 1.5), y_true), axis=-1)
